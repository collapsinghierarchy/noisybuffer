Embed NoisyBuffer

Copy the snippet below — swap YOUR-APP-ID for your own UUID:

<!-- NoisyBuffer widget -->
<script src="https://HOST_URL/nb.js"></script>
<script>
  NB.init({
    appId: "YOUR-APP-ID",          // ← replace
    // apiBase: "/api/nb/v1"       // optional
  });
</script>

<form data-noisybuffer>
  <input name="email" required placeholder="Email">
  <textarea name="message" required placeholder="Your message"></textarea>
  <button>Send</button>
</form>
